async function retryFailedExtractor(id) {
  const ui = getUiState(id);
  setPending(id, "retry", true);

  try {
    const payload = await request(`/api/extractors/${id}/retry-failed`, {
      method: "POST"
    });
    storeState(id, payload.state);
    return payload;
  } finally {
    setPending(id, "retry", false);
  }
}

14. Add this new helper function before `runLoop(id)`:

async function runTranscriptionLoop(id) {
  const ui = getUiState(id);
  if (ui.autoRun) {
    return;
  }

  ui.autoRun = true;
  setPending(id, "run", true);
  renderCurrentView();

  let processed = 0;
  let done = 0;
  let failed = 0;
  let reason = "queue-empty";

  try {
    while (ui.autoRun) {
      const payload = await request(`/api/extractors/${id}/claim-browser-task`, {
        method: "POST"
      });
      storeState(id, payload.state);

      if (!payload.task) {
        reason = payload.reason || (ui.latestState?.runtime?.stopRequested ? "stop-requested" : "queue-empty");
        break;
      }

      processed += 1;

      try {
        const result = await transcribeTaskInBrowser(payload.task);
        const completePayload = await request(`/api/extractors/${id}/complete-browser-task`, {
          method: "POST",
          body: {
            recordId: payload.task.recordId,
            sourceAdRecordId: payload.task.sourceAdRecordId,
            transcript: result.text,
            bytes: result.bytes
          }
        });
        storeState(id, completePayload.state);
        done += 1;
      } catch (error) {
        const message = error instanceof Error ? error.message : String(error);
        failed += 1;
        const fatal = isEnvironmentFailure(message);
        const failPayload = await request(`/api/extractors/${id}/fail-browser-task`, {
          method: "POST",
          body: {
            recordId: payload.task.recordId,
            key: payload.task.key,
            error: message,
            fatal
          }
        });
        storeState(id, failPayload.state);
        if (fatal) {
          reason = "environment-error";
          ui.autoRun = false;
          break;
        }
      }

      if (ui.latestState?.runtime?.stopRequested) {
        reason = "stop-requested";
        break;
      }

      await new Promise((resolve) => setTimeout(resolve, 250));
    }
  } finally {
    const finishPayload = await request(`/api/extractors/${id}/finish-browser-run`, {
      method: "POST",
      body: {
        processed,
        done,
        failed,
        reason
      }
    }).catch(() => null);

    if (finishPayload?.state) {
      storeState(id, finishPayload.state);
    }

    ui.autoRun = false;
    setPending(id, "run", false);
    renderCurrentView();
  }
}
